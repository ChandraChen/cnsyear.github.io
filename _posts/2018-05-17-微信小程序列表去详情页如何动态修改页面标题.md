### 1. 页面标题内容固定

- 示例：

```
//页面标题代码在example.json文件中的书写格式
{
    "navigationBarTitleText": "充值业务"
}
```

### 2. 随数据库返回内容动态更新页面标题

- 示例：

```
//example.js加载时动态设置页面标题
//获取应用实例
var app = getApp();
Page({
  data: {},
  onLoad: function (options) {
    var that = this;
    that.setData({
      mername: options.mername//options为页面路由过程中传递的参数
    })
    wx.setNavigationBarTitle({
      title: that.data.mername//页面标题为路由参数
    })
  },
  onReady: function () {
    // 页面渲染完成
  },
  onShow: function () {
    // 页面显示
  },
  onHide: function () {
    // 页面隐藏
  },
  onUnload: function () {
    // 页面关闭
  }
})
```
