#### 匹配特殊符号正则表达式

```

	//特殊符号正则验证
	callback:{
		message:'不能含有特殊符号',
		callback:function(val, validator){
			var pattern = new RegExp("[`~!@#$^&*()=|{}':;',\\[\\].<>/?~！@#￥……&*（）——|{}【】‘；：”“'。，、？]");
			if(pattern.test(val)){
				return false;
			}
			return true;
		}
	},
```
